#include <enet/enet.h>
#include <iostream>
#include <map>
#include <string>
#include <cstdlib>
#include <ctime>
#include <cstring> // Adaugat pentru strcpy

enum PacketType {
    PACKET_CREATE_ROOM, PACKET_ROOM_CREATED, PACKET_JOIN_ROOM, 
    PACKET_JOIN_SUCCESS, PACKET_ROOM_NOT_FOUND, PACKET_MOVE, 
    PACKET_SCORE, PACKET_GAME_OVER
};

struct NetworkPacket {
    PacketType type;
    char roomCode[10];
    int x;
    int y;
    int score;
    bool gameOver;
};

struct Room {
    ENetPeer* player1;
    ENetPeer* player2;
    bool isFull;
};

std::map<std::string, Room> activeRooms;

std::string GenerateRoomCode() {
    std::string code = "";
    for (int i = 0; i < 4; ++i) {
        code += (char)('A' + (rand() % 26)); 
    }
    return code;
}

void SendPacket(ENetPeer* peer, NetworkPacket packet) {
    if (peer == nullptr) return;
    ENetPacket* ePacket = enet_packet_create(&packet, sizeof(NetworkPacket), ENET_PACKET_FLAG_RELIABLE);
    enet_peer_send(peer, 0, ePacket);
}

int main() {
    srand(time(NULL));

    if (enet_initialize() != 0) {
        std::cerr << "Eroare la initializare ENet!\n";
        return EXIT_FAILURE;
    }

    ENetAddress address;
    address.host = ENET_HOST_ANY;
    address.port = 7777;

    ENetHost* server = enet_host_create(&address, 32, 2, 0, 0);
    if (server == nullptr) {
        std::cerr << "Eroare la crearea serverului!\n";
        return EXIT_FAILURE;
    }

    std::cout << "Serverul Central a pornit pe portul 7777!\nAstept conexiuni...\n";

    ENetEvent event;
    while (true) {
        while (enet_host_service(server, &event, 10) > 0) {
            switch (event.type) {
                case ENET_EVENT_TYPE_CONNECT:
                    std::cout << "Un jucator s-a conectat la server.\n";
                    break;

                case ENET_EVENT_TYPE_RECEIVE: {
                    NetworkPacket* receivedData = (NetworkPacket*)event.packet->data;

                    if (receivedData->type == PACKET_CREATE_ROOM) {
                        std::string newCode = GenerateRoomCode();
                        
                        Room newRoom;
                        newRoom.player1 = event.peer;
                        newRoom.player2 = nullptr;
                        newRoom.isFull = false;
                        
                        activeRooms[newCode] = newRoom;
                        
                        NetworkPacket response;
                        response.type = PACKET_ROOM_CREATED;
                        strcpy(response.roomCode, newCode.c_str());
                        
                        // Trimitem codul înapoi jucătorului!
                        SendPacket(event.peer, response);
                        
                        std::cout << "Camera noua creata: " << newCode << "\n";
                    }
                    else if (receivedData->type == PACKET_JOIN_ROOM) {
                        // Citim codul doar daca jucatorul vrea sa intre intr-o camera!
                        std::string code(receivedData->roomCode); 

                        if (activeRooms.count(code) > 0 && !activeRooms[code].isFull) {
                            activeRooms[code].player2 = event.peer;
                            activeRooms[code].isFull = true;
                            
                            NetworkPacket startPacket;
                            startPacket.type = PACKET_JOIN_SUCCESS;
                            SendPacket(activeRooms[code].player1, startPacket);
                            SendPacket(activeRooms[code].player2, startPacket);
                            
                            std::cout << "Meci inceput in camera: " << code << "\n";
                        } else {
                            NetworkPacket errorPacket;
                            errorPacket.type = PACKET_ROOM_NOT_FOUND;
                            SendPacket(event.peer, errorPacket);
                        }
                    }
                    // ... Poti adauga logica de MOVE aici mai tarziu
                    
                    enet_packet_destroy(event.packet);
                    break;
                }

                case ENET_EVENT_TYPE_DISCONNECT:
                    std::cout << "Un jucator s-a deconectat.\n";
                    break;
                    
                default:
                    break;
            }
        }
    }

    enet_host_destroy(server);
    enet_deinitialize();
    return 0;
}
